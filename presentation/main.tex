% ---------- 2. Introduction ------------------------------------------------
\begin{frame}{What we built}
  A \alert{GitHub Action} that runs on every pull request and posts \alert{one} batched
  review with inline comments and a verdict
  (\textsc{approve} / \textsc{request\_changes} / \textsc{comment}).
  \vspace{0.6em}

  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
      \begin{itemize}
        \item Open-source, runs in any repo with 3 secrets
        \item Repo-aware: indexes the codebase before reviewing
        \item Self-hosted model -- no per-PR API cost
      \end{itemize}
    \column{0.42\textwidth}
      \chip{\faGithub\ PR opened / synchronized}\\[6pt]
      \chip{\faRobot\ Agent loop, $\leq$ 20 steps (tunable)}\\[6pt]
      \chip{\faCommentDots\ 1 review, $\leq$ 10 inline comments (tunable)}
  \end{columns}
\end{frame}

% ---------- 3. Problem Definition ------------------------------------------
\begin{frame}{Problem}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \textbf{Manual review}
      \begin{itemize}
        \item Slow -- blocks merges, reviewers context-switch
        \item Inconsistent across reviewers and repos
        \item Doesn't scale with team size or PR rate
      \end{itemize}
    \column{0.48\textwidth}
      \textbf{Off-the-shelf LLM review}
      \begin{itemize}
        \item No project context -- hits only the diff
        \item Generic comments, hallucinated APIs
        \item Can't actually run tests, lint, or grep
      \end{itemize}
  \end{columns}
  \vspace{0.8em}
  \begin{block}{Goal}
    Reviews that are \alert{specific}, \alert{grounded in the repo}, and produced by an
    agent that can \alert{investigate} before commenting.
  \end{block}
\end{frame}

% ---------- 6b. Dataset ----------------------------------------------------
\begin{frame}{Dataset -- Microsoft CodeReviewer}
  \footnotesize Each example in the dataset is a real code diff from an open-source GitHub PR, paired
  with the actual comment a human reviewer wrote on that diff. We use the public
  \textit{Comment Generation} split to teach the model the same task: read a diff, write a
  review.
  \vspace{0.4em}

  \begin{columns}[T,onlytextwidth]
    \column{0.50\textwidth}
      \footnotesize\textbf{Cleaning before training}
      \begin{itemize}\scriptsize
        \item Drop comments under 10 characters (mostly noise)
        \item Drop generic acknowledgements: \textit{lgtm}, \textit{+1}, \textit{done}, \textit{nit}, \textit{ack}, \textit{fixed}
        \item Drop bare URLs and formatting-only changes
        \item Convert the dataset's \texttt{<add>}/\texttt{<del>} tags into a normal \texttt{+}/\texttt{-} line-prefix diff
      \end{itemize}
    \column{0.46\textwidth}
      \footnotesize\textbf{One example becomes a 3-turn chat}
      \begin{itemize}\scriptsize
        \item \textbf{system}: a fixed instruction (next slide)
        \item \textbf{user}: \texttt{Review this code change: ```diff ... ```}
        \item \textbf{assistant}: the human reviewer's actual comment from the PR
      \end{itemize}
  \end{columns}
  \vspace{0.6em}

  \footnotesize After cleaning we shuffle and split 80/10/10 (seed 42): \alert{$\sim$94k train},
  \alert{$\sim$11.7k val}, \alert{11.7k test}. The test split is held out -- every number in our
  results tables is computed on examples the model has never seen.
  \vspace{0.4em}
\end{frame}

% ---------- 6c. SFT implementation -----------------------------------------
\begin{frame}{SFT -- our setup}
  \footnotesize We fine-tune Qwen3.5-9B on the prepared chat-format dataset using TRL's
  \texttt{SFTTrainer} with Unsloth's kernel patches for speed. Only the LoRA adapters are
  updated; the base model stays frozen and quantised.
  \vspace{0.4em}
  \begin{columns}[T,onlytextwidth]
    \column{0.50\textwidth}
      \footnotesize\textbf{Training} \;\chip{TRL SFTTrainer + Unsloth}\;\chip{A100 80GB}
      \begin{itemize}\scriptsize
        \item Base model loaded in NF4 4-bit, bf16 compute, \texttt{max\_seq=2048}
        \item QLoRA adapters: rank $r{=}16$, $\alpha{=}16$, dropout 0
        \item Adapters attached to every projection layer (\texttt{q,k,v,o,gate,up,down\_proj})
        \item lr \texttt{2e-4}, cosine schedule, 10 warmup steps, AdamW 8-bit
        \item Per-device batch 32, weight decay 0.01, grad clip 1.0
      \end{itemize}
      \vspace{4pt}
    \column{0.46\textwidth}
      \begin{block}{\footnotesize System prompt sent with every training example}

        \vspace{0.2cm}
      
        \scriptsize\itshape You are an expert code reviewer. Given a code diff,
        provide a concise, actionable review comment. Focus on bugs, security
        vulnerabilities, performance issues, and code style. Be specific:
        reference the exact code that needs changing and explain why. If the
        code looks correct, say so briefly.
      \end{block}
  \end{columns}
  \vspace{0.4em}
\end{frame}

% ---------- 6d. GRPO our setup + reward prompt -----------------------------
\begin{frame}{GRPO -- our setup \& reward}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \textbf{Reward}\;\chip{Claude Haiku 4.5 judge}
      \begin{itemize}\scriptsize
        \item Scores each candidate review 1--5 on \emph{technical correctness} + \emph{specificity}
        \item Length penalty up to $-0.5$ over 100 words -- keeps comments tight
        \item 8 parallel judge workers, retry / backoff on rate-limits
      \end{itemize}
      \vspace{4pt}
      \textbf{Training} \;\chip{TRL GRPOTrainer + Unsloth}
      \begin{itemize}\scriptsize
        \item Start from the SFT adapter, same LoRA shape
        \item $G{=}4$ completions / prompt at $T{=}0.8$
        \item \texttt{max\_prompt=512}, \texttt{max\_completion=512}
        \item lr \texttt{2e-6}, batch 1, grad accum 4, 1 epoch, bf16
        \item \alert{250 prompts} (subset of SFT training data)
      \end{itemize}
      \vspace{4pt}
    \column{0.50\textwidth}
      \begin{block}{\footnotesize Reward prompt }

        \vspace{0.2cm}
      
        \tiny\ttfamily
        You are evaluating an automated code review comment on a code diff.\\[3pt]
        \textless code-diff\textgreater~ [DIFF] ~\textless/code-diff\textgreater\\
        \textless review-comment\textgreater~ [COMMENT] ~\textless/review-comment\textgreater\\[3pt]
        Score 1--5 on technical correctness and specificity.\\
        Length, tone, confidence do not affect the score.\\[3pt]
        5 -- real issue, exact location, explains why, concrete fix.\\
        4 -- real issue but vague about location, cause, or fix.\\
        3 -- partially correct; explanation or fix incomplete.\\
        2 -- generic; could apply to any diff.\\
        1 -- factually wrong, refers to code not in the diff, or toxic.\\[3pt]
        Respond with a single integer (1--5) and nothing else.
      \end{block}
  \end{columns}
\end{frame}

% ---------- 7. RAG --------------------------------------------------------
\begin{frame}{RAG -- indexing the repo}
  \footnotesize Before the agent runs, the action walks the checked-out repo and indexes it into
  a per-repo Chroma collection. The agent's \texttt{semantic\_search} tool queries
  that collection during the review.
  \vspace{0.4em}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \footnotesize\textbf{Ingestion}
      \begin{itemize}\scriptsize
        \item Walks the repo, skips build / cache dirs (\texttt{.git}, \texttt{node\_modules}, \texttt{.venv}, \texttt{\_\_pycache\_\_}, \texttt{dist}, \texttt{build}, \dots)
        \item Keeps \texttt{.py}, \texttt{.md}, \texttt{.ts}, \texttt{.js}, \texttt{.go}, \texttt{.java}, \texttt{.rs}
        \item \alert{Chunking}: 60-line windows with 10-line overlap; \texttt{start\_line} kept in metadata so retrieval can cite a region
        \item Sent in batches of 25 to the ingest endpoint
      \end{itemize}
    \column{0.48\textwidth}
      \footnotesize\textbf{Embeddings \& storage}
      \begin{itemize}\scriptsize
        \item Embedder: \texttt{nomic-ai/CodeRankEmbed}, run locally via \texttt{SentenceTransformer}
        \item \texttt{chromadb.PersistentClient} on disk, one collection per repository, HNSW with cosine distance
        \item Doc IDs are a SHA-256 of \texttt{(repo, path, chunk\_index, content)}; we \texttt{upsert} so re-ingesting unchanged content is a no-op
      \end{itemize}
  \end{columns}
  \vspace{0.5em}

  \scriptsize\textbf{Retrieval:} \texttt{semantic\_search(query, k)} returns the top-$k$ chunks with
  their file path and a cosine-similarity score (computed as $1 - \text{distance}$). $k$ defaults to 10,
  capped at 20.
  \vspace{0.4em}
\end{frame}

% ---------- 7b. Agent & tools ---------------------------------------------
\begin{frame}{Agent \& tools}
  \textbf{smolagents \texttt{CodeAgent}} -- writes Python tool calls, ends with
  \texttt{final\_answer("done")}.\;
  Defaults: \alert{20 steps}, \alert{10 inline comments} (both tunable via env).
  \vspace{0.3em}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \textbf{\small Retrieval}
      \begin{itemize}\scriptsize
        \item \texttt{semantic\_search}\,(CodeRankEmbed + Chroma, behind a local FastAPI service)
        \item \texttt{search\_keyword}\,(\texttt{grep -rn})
        \item \texttt{search\_symbol}\,(Python \texttt{ast}, \texttt{.py} files)
        \item \texttt{get\_file}\,(path + optional \texttt{line\_range})
      \end{itemize}
    \column{0.48\textwidth}
      \textbf{\small Context}
      \begin{itemize}\scriptsize
        \item \texttt{check\_history}\,(\texttt{git blame})
        \item \texttt{get\_pr\_metadata}\,(GitHub REST: title, author, base/head, files)
        \item \texttt{get\_team\_conventions}\,(\texttt{CLAUDE.md} / \texttt{CONTRIBUTING.md} / \texttt{.cursorrules})
        \item \texttt{load\_skill} -- playbooks (\texttt{react}, \texttt{nextjs}, \texttt{vite}, \texttt{supabase}, \texttt{async-js})
      \end{itemize}
  \end{columns}
  \vspace{0.3em}
  \textbf{\small Action} \;\textcolor{mute}{\scriptsize buffered into \texttt{ReviewState}, sent as one review}
  \begin{itemize}\scriptsize
    \item \texttt{post\_comment(file, line, severity, category, suggestion)} \;|\; \texttt{propose\_patch(file, line, diff)}
    \item Verdict (\alert{first-call-wins}): \texttt{approve} \;|\; \texttt{request\_changes} \;|\; \texttt{comment\_only}
  \end{itemize}
  \vspace{0.2em}
  \footnotesize \texttt{ReviewState} dedups by $(file, line)$, enforces the budget, and POSTs one
  \texttt{/repos/.../pulls/.../reviews} at the end -- auto-downgrades \texttt{APPROVE}\,$\rightarrow$\,\texttt{COMMENT} when GitHub returns 422.
\end{frame}

% ---------- 7c. Experiments we ran ----------------------------------------
\begin{frame}{Experiments we ran on the agent loop}
  \footnotesize Five ablations against a fixed PR set, all measured against the GEval rubric.
  None of them required retraining -- runtime / prompt / RAG changes only.
  \vspace{0.4em}

  \scriptsize
  \begin{tabular}{@{}p{0.30\textwidth} p{0.32\textwidth} p{0.30\textwidth}@{}}
    \toprule
    \textbf{What we tried} & \textbf{Setup} & \textbf{Result} \\
    \midrule
    Skills system off vs on
      & 8 PRs, \texttt{max-steps=15}, same model
      & \alert{Recall $\times$\,5}: \texttt{1/40} $\rightarrow$ \texttt{5/40} TPs \\[2pt]
    Step-budget sweep
      & PR \#11, \texttt{max-steps} $\in$ \{4, 8, 16, 24\}
      & Sweet spot at \textbf{8}; above, model bails early (\texttt{0/6} at 16, 24) \\[2pt]
    Multi-run accumulation
      & 3 PRs $\times$\,3 sequential reruns; dedup by \texttt{(file, line)}
      & PR \#5: \texttt{R} climbs \texttt{0.00 $\rightarrow$ 0.14 $\rightarrow$ 0.43} \\[2pt]
    Mandate-investigate prompt
      & Rewrote prompt: require $\geq$\,1 retrieval call before \texttt{post\_comment}
      & \texttt{semantic\_search}: \alert{0\% $\rightarrow$ 11\%} of runs; recall \texttt{+42\%} relative \\[2pt]
    Chunked RAG ingestion
      & File-level chunks $\rightarrow$ 60-line windows, 10-line overlap
      & Cross-file riddle PR \#53 catchable: agent reads \emph{both} files \\
    \bottomrule
  \end{tabular}
  \vspace{6pt}

  \footnotesize\textbf{What stuck:} \scriptsize Shipped \texttt{skills-on}, \texttt{steps=20--30},
  \texttt{mandate-investigate} prompt, chunked RAG. Multi-run is a free recall multiplier any
  consumer can adopt.
  \vspace{0.4em}
\end{frame}

% ---------- 8c. Agentic eval -- setup + summary ---------------------------
\begin{frame}{Agentic evaluation -- setup \& metrics}
  \textbf{Setup} \;\chip{4-bit GGUF agent, local}\;\chip{judge: gemma-4-e4b-it}\;\chip{15 ex, max 20 steps}
  \vspace{0.5em}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \textbf{Three metrics}
      \begin{itemize}\scriptsize
        \item \textbf{Task completion} (programmatic) -- 1.0 if a valid verdict was set before the step budget ran out, else 0.0
        \item \textbf{Task correctness} (LLM-judged, Gemma GEval) -- 0--1, aligned with the RLHF reward rubric
        \item \textbf{Step efficiency} (programmatic) -- decays linearly above the 10-call cap; 0.3 for verdict-with-no-investigation; 0 for no verdict
      \end{itemize}
      \vspace{6pt}
      \begin{block}{\footnotesize GEval rubric for correctness}
        \scriptsize
        Is the flagged issue real and present in the diff?\\
        Does the comment pinpoint the exact location (file, line)?\\
        Does it explain \emph{why} the code is problematic?\\
        Does it propose a concrete, actionable fix?\\
        Is it specific to this diff, not a generic observation?
      \end{block}
    \column{0.48\textwidth}
      \footnotesize
      \begin{tabular}{@{}lccc@{}}
        \toprule
        Metric & Mean & Pass rate & Threshold \\
        \midrule
        Task completion   & 0.85 & 13/15 (86.7\%) & $\geq$ 0.50 \\
        Task correctness  & 0.85 & 13/15 (86.7\%) & $\geq$ 0.50 \\
        Step efficiency   & 0.82 & 12/15 (80.0\%) & $\geq$ 0.50 \\
        \bottomrule
      \end{tabular}
      \vspace{8pt}

      \textbf{What ``step efficiency'' means}
      \begin{itemize}\scriptsize
        \item Best score: 1--2 tool calls and a valid verdict
        \item 0.3 floor for the \emph{verdict with 0 tool calls} case -- a valid verdict without any investigation
        \item 0 for runs that never reach a verdict
      \end{itemize}
  \end{columns}
\end{frame}
