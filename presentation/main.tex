% ---------- 2. Introduction ------------------------------------------------
\begin{frame}{What we built}
  A \alert{GitHub Action} that runs on every pull request and posts \alert{one} batched
  review with inline comments and a verdict
  (\textsc{approve} / \textsc{request\_changes} / \textsc{comment}).
  \vspace{0.6em}

  \begin{columns}[T,onlytextwidth]
    \column{0.55\textwidth}
      \begin{itemize}
        \item Open-source, runs in any repo with 3 secrets
        \item Repo-aware: indexes the codebase before reviewing
        \item Self-hosted model -- no per-PR API cost
      \end{itemize}
    \column{0.42\textwidth}
      \chip{\faGithub\ PR opened / synchronized}\\[6pt]
      \chip{\faRobot\ Agent loop, $\leq$ 20 steps}\\[6pt]
      \chip{\faCommentDots\ 1 review, $\leq$ 10 inline comments}
  \end{columns}
\end{frame}

% ---------- 3. Problem Definition ------------------------------------------
\begin{frame}{Problem}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \textbf{Manual review}
      \begin{itemize}
        \item Slow -- blocks merges, reviewers context-switch
        \item Inconsistent across reviewers and repos
        \item Doesn't scale with team size or PR rate
      \end{itemize}
    \column{0.48\textwidth}
      \textbf{Off-the-shelf LLM review}
      \begin{itemize}
        \item No project context -- hits only the diff
        \item Generic comments, hallucinated APIs
        \item Can't actually run tests, lint, or grep
      \end{itemize}
  \end{columns}
  \vspace{0.8em}
  \begin{block}{Goal}
    Reviews that are \alert{specific}, \alert{grounded in the repo}, and produced by an
    agent that can \alert{investigate} before commenting.
  \end{block}
\end{frame}

% ---------- 6. Training ----------------------------------------------------
\begin{frame}{Training -- SFT then GRPO}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
      \textbf{\faLayerGroup\ SFT} \;\chip{1 epoch}\;\chip{A100 80GB}
      \begin{itemize}\small
        \item Base: Qwen3.5-9B, 4-bit NF4, bf16
        \item Data: Microsoft CodeReviewer, filtered noisy comments (\textit{lgtm}, \textit{+1}, urls, \textless10 chars)
        \item QLoRA $r{=}16$, $\alpha{=}16$, all proj layers
        \item lr 2e-4, batch 32, cosine, AdamW 8-bit
      \end{itemize}
    \column{0.48\textwidth}
      \textbf{\faTrophy\ GRPO (RLHF)} \;\chip{TRL}
      \begin{itemize}\small
        \item 4 samples / prompt, group-relative reward
        \item Reward: Claude Haiku 4.5 judge, 1--5 on correctness + specificity
        \item Length penalty up to $-0.5$ over 100 words
        \item 250 prompts, lr 2e-6, bf16
      \end{itemize}
  \end{columns}
  \vspace{0.4em}
  \footnotesize Published: \texttt{cristicretu/code-reviewer-lora} (SFT),
  \texttt{cretu-luca/code-reviewer-grpo} (RLHF),
  \texttt{cretu-luca/code-reviewer-grpo-GGUF} (4-bit, 5.2\,GB).
\end{frame}

% ---------- 7. Agent & RAG -------------------------------------------------
\begin{frame}{Agent \& tools}
  \textbf{smolagents \texttt{CodeAgent}} -- writes Python tool calls,
  budget locked at \alert{20 steps} and \alert{10 comments}.
  \vspace{0.5em}
  \begin{columns}[T,onlytextwidth]
    \column{0.32\textwidth}
      \textbf{\small Retrieval}
      \begin{itemize}\scriptsize
        \item \texttt{semantic\_search}\\(CodeRankEmbed + Chroma)
        \item \texttt{search\_keyword} (rg)
        \item \texttt{search\_symbol} (AST)
        \item \texttt{get\_file}
      \end{itemize}
    \column{0.32\textwidth}
      \textbf{\small Execution}
      \begin{itemize}\scriptsize
        \item \texttt{run\_tests} (pytest)
        \item \texttt{run\_linter} (ruff)
        \item \texttt{run\_typecheck} (mypy)
        \item \texttt{run\_snippet} (sandbox, 5\,s)
      \end{itemize}
    \column{0.32\textwidth}
      \textbf{\small Context / action}
      \begin{itemize}\scriptsize
        \item \texttt{check\_history} (blame)
        \item \texttt{load\_skill} -- on-demand playbooks
        (\texttt{react}, \texttt{nextjs}, \texttt{vite}, \texttt{supabase}, \texttt{async-js})
        \item \texttt{post\_comment}, \texttt{propose\_patch}
        \item verdict tools
      \end{itemize}
  \end{columns}
  \vspace{0.4em}
  \footnotesize \texttt{ReviewState} dedups $(file, line)$ comments and batches \alert{one}
  \texttt{POST /reviews} at the end.
\end{frame}

      \vspace{6pt}

      \scriptsize
      \textbf{Agentic pipeline (15 ex):} \\
      Task completion \textbf{86.7\%} \;|\; Correctness \textbf{86.7\%}
      \;|\; Step efficiency \textbf{80.0\%}

    \column{0.42\textwidth}
      \textbf{Takeaways}
      \begin{itemize}\small
        \item Base model is unusable (empty outputs)
        \item SFT closes most of the gap
        \item GRPO: marginal on auto metrics; Gemini judge \alert{prefers SFT}
        \item Reward over-fit to Haiku $\Rightarrow$ a single-judge GRPO setup
          is a real risk
      \end{itemize}
  \end{columns}
\end{frame}

% ---------- 11. Conclusion -------------------------------------------------
\begin{frame}{Wrap-up}
  \textbf{What works}
  \begin{itemize}\small
    \item Repo-aware agent reaches a valid verdict on 87\% of held-out PRs
    \item Q4\_K\_M GGUF runs on a 24\,GB M4 -- no GPU bill for the demo
    \item Composite Action drops into any repo in $\sim$10 lines of YAML
  \end{itemize}

  \textbf{What we'd improve next}
  \begin{itemize}\small
    \item Multi-judge reward (rotate Haiku / Gemini / Gemma) to avoid over-fit
    \item Larger agentic eval set, with human spot-checks
    \item Better dedup against past review threads on the same file
  \end{itemize}
